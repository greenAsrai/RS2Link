\section{Labelled Transition System for Reaction Systems}
\label{sec:LTSforRS}

\begin{definition}\label{def:LTSforRS}
\[
\begin{array}{lcl}
S&::=&[K]\\
K&::=&(R,I,P)~\mid~\{e\}~\mid~K|K~\mid C\\
C&::=&recX.C~\mid~X~\mid~E.C~\mid~C+C\\
\end{array}
\]
We assume that $\{e_1\}|\{e_2\}= \{e_1,e_2\}$
\end{definition}
